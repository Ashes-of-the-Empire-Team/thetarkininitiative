	if = {
		limit = {
			has_political_power = 100
			check_variable = {currency_amount^primary_currency > 150} ## ACTUAL TRIGGER TO DECIDE WHAT COST OF THE TILE BLOCKER IS
		}
		subtract_from_variable = { currency_amount^primary_currency = 150 }
		add_political_power = -100

		## BELOW IS ALWAYS NEEDED
		set_temp_variable = {building_name_passer = building_1}  ### THIS IS WHATS ACTUALLY REMOVING THE BUILDING
		clear_building_1_effects = yes
	}

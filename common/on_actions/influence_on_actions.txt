on_actions = {
	#### EMPIRE ON ACTIONS FOR LEGITIMACY ####
	on_startup = {
		effect = {
			every_country = {
				if = {
					limit = {
						NOT = {
							has_country_flag = major_influencial_nation
						}
						capital_scope = {
							is_on_continent = hutt_space
						}
					}
					add_to_array = { Global.the_slice_array = THIS.id }
				}
			}
		}
	}
}

							set_variable = {ship_counting_to_zero_holder@var:heavy_ship_selected_item = ship_amount_contract_holder@var:heavy_ship_selected_item^1}
						}
					}
				}else = {
					set_temp_variable = {heavy_ship_calculator = heavy_ship_workers}
					divide_temp_variable = {heavy_ship_calculator = 100}
					subtract_from_variable = {heavy_selected_ship_to_be_reduced = heavy_ship_calculator}
				}
			}
		}
